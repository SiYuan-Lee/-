class Solution {
    public boolean isValidSudoku(char[][] board) {
       HashMap<Integer,Integer>  []mars= new HashMap[9];
       HashMap<Integer,Integer>  []earth= new HashMap[9];
       HashMap<Integer,Integer>  []moon= new HashMap[9];
        for(int i=0;i<9;i++){
            mars[i]=new HashMap<Integer,Integer>();
            earth[i]=new HashMap<Integer,Integer>();
            moon[i]=new HashMap<Integer,Integer>();
        }
       for(int i=0;i<9;i++){
           for(int j=0;j<9;j++){
               int index=(i/3)*3+j/3;
               if(board[i][j]!='.'){
                   int n=(int)board[i][j];
                   mars[i].put(n,mars[i].getOrDefault(n,0)+1);
                   earth[j].put(n,earth[j].getOrDefault(n,0)+1);
                   moon[index].put(n,moon[index].getOrDefault(n,0)+1);
                   if(mars[i].get(n)>1||earth[j].get(n)>1||moon[index].get(n)>1){
                       return false;
                   }
               }                              
           }           
       }
       return true;      
    }
}
