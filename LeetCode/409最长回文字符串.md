class Solution {
    public int longestPalindrome(String s) {
        int [] cnt= new int[58];//声明数组，使用关键字new
        for(char c:s.toCharArray()){//优雅的写法，不需要求字符串的长度
            cnt[c-'A']+=1; //A的asc码是65，a的asc码是97

        }
        int ans=0;
        for(int x:cnt){
            ans+=x-(x&1);//与运算的优先级很低，低于减法
        }
        return ans==s.length()?ans:ans+1;
    }
}
